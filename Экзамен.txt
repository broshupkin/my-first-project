import java.util.Scanner;

public class Main1 {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        final int DAILY_PRICE = 1800;
        final int MONTHLY_PRICE = 19900;

        System.out.println("Выберите тип абонемента: ");
        System.out.println("1 - дневной абонимент(1800 тг.) ");
        System.out.println("2 - месячный абонимент(19900 тг.) ");
        System.out.println("Введите номер варианта: ");

        int choice = scanner.nextInt();

        switch (choice) {
            case 1:
                System.out.println("Вы выбрали дневной абонемент. Стоимость: " + DAILY_PRICE + " руб.");
                break;
            case 2:
                System.out.println("Вы выбрали месячный абонемент. Стоимость: " + MONTHLY_PRICE + " руб.");
                break;
            default:
                System.out.println("Некорректный выбор. Пожалуйста, выберите 1 или 2.");
        }

        scanner.close();
    }
}
